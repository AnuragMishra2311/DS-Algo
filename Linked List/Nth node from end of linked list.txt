int reverse(Node *head){
    int length;
    if(head==NULL){
        return 0;
    }
    
    else{
        length=1+reverse(head->next);
    }
    return length;
}

int getNthFromLast(Node *head, int n)
{
       // Your code here
       Node *temp=head;
       int length=reverse(temp);
       int check=length+1-n,counter=1;
       while(head && counter<=check){
           if(counter==check){
               return head->data;
           }
           head=head->next;
           counter+=1;
       }
       return -1;
}
