void removeLoop(Node* head)
    {
        // code here
        // just remove the loop without losing any nodes
        unordered_set<Node*> s;
        while(head){
            s.insert(head);
            if(s.find(head->next)!=s.end()){
                head->next=NULL;
            }
            head=head->next;
            
        }
    }
