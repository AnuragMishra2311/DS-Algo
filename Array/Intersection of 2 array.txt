arr1=list(map(int,input().split()))
arr2=list(map(int,input().split()))

i=0
j=0


while True:
    
    try:
    
        if arr1[i]<arr2[j]:
            i+=1
        
        elif arr1[i]>arr2[j]:
            j+=1
    
        elif arr1[i]==arr2[j]:
            print(arr1[i],end=' ')
            i+=1
            j+=1
    except:
        break
