int getCount(Node *root, int l, int h)
{
  // your code goes here   
  if(root==NULL){
      return 0;
  }
  else if(l<=root->data && h>=root->data){
      return 1+ getCount(root->left,l,h) + getCount(root->right,l,h);
  }
  else{
      return getCount(root->left,l,h) + getCount(root->right,l,h);
  }
}
